// this class describes the properties of a single clocks.
 class Particle { 
// setting the co-ordinates, radius and the 
// speed of a particle in both the co-ordinates axes. 
constructor(){ 
this.x = random(0,width);
this.y = random(0,height); 
this.r = random(70,98); 
this.xSpeed = random(-1,1);
 this.ySpeed = random(-2,1.5); 
} 
// creation of a clocks. 
createParticle() {
noStroke();
fill('rgba(10,19,19,45)');//color 
circle(this.x,this.y,this.r); 
} 
// setting the clocks in motion. 
moveParticle() {
if(this.x < 0 || this.x > width) 
     this.xSpeed*=-1; 
if(this.y < 0 || this.y > height)
         this.ySpeed*=-1; 
this.x+=this.xSpeed; 
this.y+=this.ySpeed; 
}
 // this function creates the connections(lines) 
// between clocks which are less than a certain distance apart joinParticles(particles) {
       particles.forEach(element =>{ let dis = dist(this.x,this.y,element.x,element.y); if(dis<95) { stroke('rgba(95,5,25,4,65)'); 
line(this.x,this.y,element.x,element.y); 
} 
}); 
} 
}
